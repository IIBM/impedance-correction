% Gráficos comunes:
% \figuregraph{escala}{nombre/etiqueta}{pie de imagen}
\newcommand{\figuregraph}[3]{
    \begin{figure}[H]
        \center
        \includegraphics[scale=#1]{#2}
        \caption{#3}\label{#2}
    \end{figure}
}

        \endminipage
        \minipage{0.5\textwidth}
            \center
            \includegraphics[scale=#4]{#5}
            \caption{#6}\label{#5}
        \endminipage
    \end{figure}
